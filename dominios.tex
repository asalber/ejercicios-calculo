\newproblem*{dom-1}{gen}{}
%ENUNCIADO
{Hallar el dominio de las funciones siguientes:
\begin{multicols}{2}
\begin{enumerate}
  \item $f(x)=\dfrac{x^2+x+1}{x^3-x}$.

  \item $g(x)=\sqrt{x^4-1}$.

  \item $h(x)=\cos\left(\dfrac{x+3}{x^2+1}\right)$.

   \item $l(x)=\arcsen\left(\dfrac{x}{1+x}\right)$.
\end{enumerate}
\end{multicols}
}
